  \frametitle{Normalization}
      \begin{align}
      \epsilon_{cteq6.6}& \equiv \frac{n_{cteq6.6}^{after}}{n_{cteq6.6}^{before}} \\
      \epsilon_{cteq6L1}& =\epsilon_{cteq6.6} \cdot \left( \frac{\epsilon_{cteq6L1}}{\epsilon_{cteq6.6}} \right) \\
                        & =\left(\frac{n_{cteq6.6}^{after}}{n_{cteq6.6}^{before}} \right) \cdot \left( \frac{\epsilon_{cteq6L1}}{\epsilon_{cteq6.6}} \right) \\
                        & =\frac{n_{cteq6.6}^{after}}{n_{cteq6.6}^{before} \left(\frac{\epsilon_{cteq6.6}}{\epsilon_{cteq6L1}} \right)} = \epsilon_{cteq6.6}^{normalized}    
      \end{align}
